28      public KaiserEngine() {
29          this.population = BigDecimal.valueOf(95L);
30          this.zYear = 0;
31          this.yield = BigDecimal.valueOf(3L);
32          this.supplies = BigDecimal.valueOf(2800L);
33          this.humans = BigDecimal.valueOf(3000L);
34          this.area = this.humans.divide(this.yield, RoundingMode.HALF_UP);
35          this.increase = BigDecimal.valueOf(5L);
36          this.deathToll = BigDecimal.ZERO;
37          this.percentDeathToll = BigDecimal.ZERO;
38          this.deathTollSum = BigDecimal.ZERO;
39          this.externalDamage = this.humans.subtract(this.supplies);

42      public void startNewRound() {
43          this.area = this.humans.divide(this.yield, 0, RoundingMode.HALF_UP);
44          this.externalDamage = this.humans.subtract(this.supplies);
45          this.zYear++;
46          this.population = this.population.add(this.increase);
47  
48          processFamine();
49          this.cost = getRandomNumberUntil(10);
50          this.yield = cost.add(BigDecimal.valueOf(17L));
51      }

53      BigDecimal getRandomNumberUntil(int threshold) {
54          return BigDecimal.valueOf(new Random().nextInt(threshold + 1) + 1);
55      }

57      public void processFamine() {
58          if (is(q).lessThan(BigDecimal.ZERO)) {
59              this.population = this.population.divide(BigDecimal.valueOf(2L), 0, RoundingMode.HALF_UP);
60              System.out.println(KaiserEnginePrinter.ORANGE);
61              System.out.println("Eine fürchterliche Seuche hat die halbe Stadt dahingerafft!");
62              System.out.println(KaiserEnginePrinter.ANSI_RESET);
63          }
64          refreshFamineQuotient();
65      }

70      }

95      public long buyLand(Long buy) {
96          if (buy < 0) {
97              System.out.println(KaiserEnginePrinter.ANSI_PURPLE + "Ignoriere negative Eingaben - Du willst mich wohl verkackeiern." + KaiserEnginePrinter.ANSI_RESET);
98          }
99  
100         if (buy > 0) {
101             if (is(this.yield.multiply(BigDecimal.valueOf(buy))).lessThanOrEqualTo(this.supplies)) {
102                 this.area = this.area.add(BigDecimal.valueOf(buy));
103                 this.supplies = this.supplies.subtract(this.yield.multiply(BigDecimal.valueOf(buy)));
104                 this.cost = BigDecimal.ZERO; // price is recalculated per round
105             } else {
106                 throw new IllegalArgumentException("Not Enough Supplies");
107             }
108         }
109         return buy;
110     }

112     public void sellLand(Long sell) {
113         if (sell < 0) {
114             System.out.println(KaiserEnginePrinter.ANSI_PURPLE + "Ignoriere negative Eingaben - Du willst mich wohl verkackeiern." + KaiserEnginePrinter.ANSI_RESET);
115             return;
116         }
117 
118         if (is(BigDecimal.valueOf(sell)).lessThan(this.area)) {
119             this.area = this.area.subtract(BigDecimal.valueOf(sell));
120             this.supplies = this.supplies.add(this.yield.multiply(BigDecimal.valueOf(sell)));
121             this.cost = BigDecimal.ZERO; // price is recalculated per round
122         } else {
123             throw new IllegalArgumentException("Not Enough Land");
124         }
125     }

127     public void feedToPopulation(Long feed) {
128         if (feed < 0) {
129             System.out.println(KaiserEnginePrinter.ANSI_PURPLE + "Ignoriere negative Eingaben - Du willst mich wohl verkackeiern." + KaiserEnginePrinter.ANSI_RESET);
130             return;
131         }
132 
133         if (feed != 0) {
134             if (is(BigDecimal.valueOf(feed)).lessThanOrEqualTo(this.supplies)) {
135                 this.supplies = this.supplies.subtract(BigDecimal.valueOf(feed));
136                 this.cost = BigDecimal.ONE; // price is recalculated per round
137             } else {
138                 throw new IllegalArgumentException("Not Enough in Stock");
139             }
140         }

143     public void cultivate(Long cultivate) {
144         if (cultivate == 0) {
145             calculateNewPrice();
146             return;
147         }
148 
149         if (cultivate < 0) {
150             System.out.println(KaiserEnginePrinter.ANSI_PURPLE + "Ignoriere negative Eingaben - Du willst mich wohl verkackeiern." + KaiserEnginePrinter.ANSI_RESET);
151             return;
152         }
153 
154         if (is(this.area).lessThan(BigDecimal.valueOf(cultivate))) {
155             throw new IllegalArgumentException("You cannot cultivate more area than you have.");
156         }
157 
158         BigDecimal halfCultivate = BigDecimal.valueOf(cultivate).divide(BigDecimal.valueOf(2L), 0, RoundingMode.HALF_UP);
159         if (is(this.supplies).lessThan(halfCultivate)) {
160             throw new IllegalArgumentException("You cannot cultivate more than you have.");
161         }
162 
163         if (is(BigDecimal.valueOf(cultivate)).greaterThan(getPopulation().multiply(BigDecimal.TEN))) {
164             throw new IllegalArgumentException("Not enough workers available.");
165         }
166 
167         // perform seeding
168         this.supplies = this.supplies.subtract(halfCultivate);
169         calculateNewPrice();
170 
171         // yields after cultivation and population increase
172         this.yield = this.cost;
173         this.humans = this.yield.multiply(BigDecimal.valueOf(cultivate));
174 
175         // cultivation kills rats ;)
176         this.externalDamage = BigDecimal.ZERO;
177         calculateNewPrice();
178 
179         // but add some external damage in some cases in a naiive manner
180         // original condition stated: if int(c/2) <> c/2
181         if (this.cost.divide(BigDecimal.valueOf(2L), 0, RoundingMode.DOWN).intValue() == this.cost.divide(BigDecimal.valueOf(2L), 0, RoundingMode.UP).intValue()) {
182             this.externalDamage = this.supplies.divide(this.cost, 0, RoundingMode.HALF_UP);
183         }
184         this.supplies = this.supplies.subtract(this.externalDamage).add(this.humans);
185         calculateNewPrice();
186     }

197     public void finishRoundAfterActions() {
198         BigDecimal factor = BigDecimal.valueOf(20L).multiply(this.area).add(this.supplies);
199         this.increase = cost.multiply(factor).divide(this.population, 0, RoundingMode.HALF_UP).divide(BigDecimal.valueOf(100).add(BigDecimal.ONE), 0, RoundingMode.HALF_UP);
200 
201         this.cost = this.q.divide(BigDecimal.valueOf(20L), 0, RoundingMode.HALF_UP);
202         refreshFamineQuotient();
203 
204         if (is(this.population).lessThan(this.cost)) {
205             this.deathToll = BigDecimal.ZERO;
206             return; // start new round without any deaths
207         }
208 
209         // calculate deaths
210         this.deathToll = this.population.subtract(this.cost);
211         if (is(this.deathToll).greaterThan(this.population.multiply(BigDecimal.valueOf(0.45)))) {
212             System.out.println(KaiserEnginePrinter.ANSI_YELLOW);
213             System.out.println("Sie haben " + this.deathToll + " Menschen in nur einem Jahr verhungern lassen!");
214             System.out.println("Auf Grund dieser extremen Misswirtschaft, werden Sie nicht nur aus Amt und Würden gejagt,");
215             System.out.println("sondern auch zum Versager des Jahres erklärt.");
216             System.out.println(KaiserEnginePrinter.ANSI_RESET);
217             return; // TODO stop the game here!
218         }
219 
220         // calc death statistics
221         // p1 = ((z-1)*p1+D*100/p)/z
222         BigDecimal tempQuotient = this.percentDeathToll.multiply(BigDecimal.valueOf(this.zYear - 1)).add(this.deathToll.multiply(BigDecimal.valueOf(100)).divide(this.population, 0, RoundingMode.HALF_UP));
223         this.percentDeathToll = tempQuotient.divide(BigDecimal.valueOf(this.zYear), 0, RoundingMode.HALF_UP);
224 
225         this.population = this.cost; // TODO why? shouldn't this somehow be added up?
226         this.deathTollSum = this.deathTollSum.add(this.deathToll);
227     }

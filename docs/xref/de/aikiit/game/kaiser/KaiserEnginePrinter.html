10  public class KaiserEnginePrinter {
11  
12      // inspired by https://talyian.github.io/ansicolors/
13      public static final String ANSI_RESET = "\u001B[0m";
14      public static final String ANSI_BLACK = "\u001B[30m";
15      public static final String ANSI_RED = "\u001B[31m";
16      public static final String ANSI_GREEN = "\u001B[32m";
17      public static final String ANSI_YELLOW = "\u001B[33m";
18      public static final String ANSI_BLUE = "\u001B[34m";
19      public static final String ANSI_PURPLE = "\u001B[35m";
20      public static final String ANSI_CYAN = "\u001B[36m";
21      public static final String ANSI_WHITE = "\u001B[37m";
22  
23      public static final String ORANGE = "\u001B[38;5;202m";
24  
25      private final KaiserEngine engine;
26  
27      public KaiserEnginePrinter(KaiserEngine engine) {
28          this.engine = engine;
29      }
30  
31      public String getStatus(int round) {
32          StringBuilder status = new StringBuilder();
33  
34          status.append(ANSI_GREEN).append("### STATUS nach Runde ").append(round).append(ANSI_RESET).append(System.lineSeparator());
35          status.append(ANSI_CYAN).append("Die Einwohnerzahl beträgt jetzt   ").append(StringUtils.leftPad(String.valueOf(this.engine.getPopulation()), 10)).append(ANSI_RESET).append(System.lineSeparator());
36          status.append(ANSI_CYAN).append("Die Stadt besitzt an Land (Hektar)").append(StringUtils.leftPad(String.valueOf(this.engine.getArea()), 10)).append(ANSI_RESET).append(System.lineSeparator());
37          status.append(ANSI_CYAN).append("Die Ernte betrug (dzt/Hektar)     ").append(StringUtils.leftPad(String.valueOf(this.engine.getYield()), 10)).append(ANSI_RESET).append(System.lineSeparator());
38          status.append(ANSI_RED).append("Ratten haben gefressen (dzt)      ").append(StringUtils.leftPad(String.valueOf(this.engine.getExternalDamage()), 10)).append(ANSI_RESET).append(System.lineSeparator());
39          status.append(ANSI_RED).append("Eure Vorräte betragen (dzt)       ").append(StringUtils.leftPad(String.valueOf(this.engine.getSupplies()), 10)).append(ANSI_RESET).append(System.lineSeparator());
40          return status.toString();
41      }
42  
43      public String getYearResult(int round) {
44          StringBuilder status = new StringBuilder();
45  
46          status.append(ANSI_GREEN).append("Im Jahr ").append(round).append(ANSI_RESET).append(System.lineSeparator());
47          status.append(ANSI_CYAN).append("Es starben         ").append(StringUtils.leftPad(String.valueOf(this.engine.getDeathToll()), 10)).append(" Einwohner").append(ANSI_RESET).append(System.lineSeparator());
48          status.append(ANSI_CYAN).append("hinzugekommen sind ").append(StringUtils.leftPad(String.valueOf(this.engine.getIncrease()), 10)).append(" Einwohner").append(ANSI_RESET).append(System.lineSeparator());
49          return status.toString();
50  
51      }
52  
53      public String getResults() {
54          StringBuilder result = new StringBuilder();
55  
56          result.append("In Ihrer ").append(KaiserGame.MAX_ROUNDS).append("-jährigen Amtszeit sind durchschnittlich ");
57          result.append(this.engine.getPercentDeathToll()).append(" % im Jahr verstorben.").append(System.lineSeparator());
58          result.append("Insgesamt sind damit ").append(this.engine.getDeathTollSum()).append(" Einwohner verstorben.").append(System.lineSeparator());
59  
60          result.append(System.lineSeparator());
61          result.append("Anfangs hatten Sie 10 Hektar pro Einwohner, jetzt haben Sie ").append(this.engine.getAreaPerCapita()).append(" Hektar/Einwohner.").append(System.lineSeparator());
62  
63          return result.toString();
64      }

66      public String evaluateRegency() {
67          StringBuilder result = new StringBuilder();
68          result.append(ANSI_BLUE).append(">>>> Bewertung Ihrer Herrschaft <<<<").append(System.lineSeparator());
69  
70          result.append("In ihrer " + engine.getZYear() + "-jährigen Amtszeit sind ");
71          result.append(engine.getPercentDeathToll() + "% der Bevölkerung im Jahr verstorben.").append(System.lineSeparator());
72          result.append("Insgesamt sind ").append(engine.getDeathTollSum()).append(" Einwohner gestorben.").append(System.lineSeparator());
73  
74          BigDecimal legacy = engine.getArea().divide(engine.getPopulation(), RoundingMode.HALF_UP);
75          result.append("Zu Beginn hatten sie 10 Hektar pro Einwohner,").append(System.lineSeparator());
76          result.append("jetzt sind es ").append(legacy).append(" Hektar/Einwohner.").append(System.lineSeparator());
77  
78          // TODO: check and fix legacy calculation - message should match criteria
79          if (is(engine.getPercentDeathToll()).greaterThan(BigDecimal.valueOf(33)) || is(legacy).lessThan(BigDecimal.valueOf(7))) {
80              result.append("Auf Grund dieser extremen Misswirtschaft, werden Sie nicht nur aus Amt und Würden gejagt,").append(System.lineSeparator());
81              result.append("sondern auch zum Versager des Jahres erklärt.").append(System.lineSeparator());
82          } else {
83  
84              if (is(engine.getPercentDeathToll()).greaterThan(BigDecimal.TEN) || is(legacy).lessThan(BigDecimal.valueOf(9))) {
85                  result.append("Ihre hartherzige Regierungsmethode erinnert an Nero und Iwan den Schrecklichen.");
86                  result.append(System.lineSeparator());
87                  result.append("Die (verbliebenen) Einwohner würden Sie gerne zum Teufel jagen!");
88              } else if (is(engine.getPercentDeathToll()).greaterThan(BigDecimal.TEN) || is(legacy).lessThan(BigDecimal.valueOf(9))) {
89                  result.append("Sie hätten glücklicher agieren können, aber es war nicht übel.");
90                  result.append(System.lineSeparator());
91                  result.append(this.engine.getPopulation().multiply(new BigDecimal("0.8"))).append(" Einwohner möchten Sie zwar hängen sehen, aber kleine Probleme hat ja jeder.");
92              } else {
93                  result.append("Eine wunderbare Regierungszeit! Karl, Konrad und Willi (die Großen) hätten es nicht besser machen können.");
94              }
95          }
96  
97          result.append(ANSI_RESET);
98          result.append(System.lineSeparator());
99          result.append(System.lineSeparator());
100         return result.toString();
101     }
